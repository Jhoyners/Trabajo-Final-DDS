CREATE DATABASE Nequi
USE Nequi


CREATE TABLE Users (
	PIN INT PRIMARY KEY NOT NULL,
	Name Varchar(25) NOT NULL,
	Last_Name VARCHAR(25) NOT NULL,
	User_ID INT NOT NULL,
);

CREATE TABLE Withdrawals (
    PIN INT NOT NULL,
    Account_Number INT PRIMARY KEY NOT NULL,
    Value DECIMAL(10,2) NOT NULL,
    Date_Time DATETIME NOT NULL,
    FOREIGN KEY (PIN) REFERENCES Users(PIN)
);

CREATE TABLE Assists (
    Phone INT PRIMARY KEY NOT NULL,
    Assist_Date DATE NOT NULL,
    Account_Number INT NOT NULL,
	Address VARCHAR(20),
    FOREIGN KEY (Account_Number) REFERENCES Withdrawals(Account_Number)
);

CREATE TABLE Complaints (
    Complaint_ID INT PRIMARY KEY IDENTITY(1,1),
    Account_Number INT NOT NULL,
    Description VARCHAR(255) NOT NULL,
    FOREIGN KEY (Account_Number) REFERENCES Withdrawals(Account_Number)
);

-- Aqui estan los CRUD

CREATE PROCEDURE DeleteUser 
    @PIN INT
AS
BEGIN
    DELETE FROM Users WHERE PIN = @PIN;
END;


CREATE PROCEDURE UpdateUser 
    @PIN INT,
    @Name VARCHAR(25),
    @Last_Name VARCHAR(25),
    @User_ID INT
AS
BEGIN
    UPDATE Users
    SET Name = @Name, Last_Name = @Last_Name, User_ID = @User_ID
    WHERE PIN = @PIN;
END;


CREATE PROCEDURE UpdateWithdrawal
    @PIN INT,
    @Account_Number INT,
    @Value DECIMAL(10,2),
    @Date_Time DATETIME
AS
BEGIN
    UPDATE Withdrawals
    SET Value = @Value, Date_Time = @Date_Time
    WHERE PIN = @PIN AND Account_Number = @Account_Number;
END;

CREATE PROCEDURE DeleteWithdrawal
    @PIN INT,
    @Account_Number INT
AS
BEGIN
    DELETE FROM Withdrawals WHERE PIN = @PIN AND Account_Number = @Account_Number;
END;


CREATE PROCEDURE UpdateAssist
    @Phone INT,
    @Assist_Date DATE,
    @Account_Number INT,
    @Address VARCHAR(20)
AS
BEGIN
    UPDATE Assists
    SET Assist_Date = @Assist_Date, Account_Number = @Account_Number, Address = @Address
    WHERE Phone = @Phone;
END;

CREATE PROCEDURE DeleteAssist
    @Phone INT
AS
BEGIN
    DELETE FROM Assists WHERE Phone = @Phone;
END;

CREATE PROCEDURE UpdateComplaint
    @Complaint_ID INT,
    @Account_Number INT,
    @Description VARCHAR(255)
AS
BEGIN
    UPDATE Complaints
    SET Account_Number = @Account_Number, Description = @Description
    WHERE Complaint_ID = @Complaint_ID;
END;

CREATE PROCEDURE DeleteComplaint
    @Complaint_ID INT
AS
BEGIN
    DELETE FROM Complaints WHERE Complaint_ID = @Complaint_ID;
END;
